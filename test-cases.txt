```maml
// ============================================================
// BASIC IMMUTABILITY
// ============================================================

x := 10
x = 20 // ERROR: immutable binding


mut x := 10
x = 20 // OK


// ============================================================
// COPY TYPES
// ============================================================

x := 10
y := x // copy
print(x) // OK
print(y) // OK


mut x := 10
y := x // copy
x = 20 // OK
print(y) // still 10


Point :: struct {
    x int
    y int
}

p := Point{x:1, y:2}
q := p // copy
print(p.x) // OK
print(q.x) // OK


// ============================================================
// REFERENCE TYPES
// ============================================================

v := Vec<int>()
w := v // shared immutable alias

v.push(1) // ERROR: immutable binding
w.push(1) // ERROR: immutable binding


mut v := Vec<int>()
v.push(1) // OK


// ============================================================
// SHARED IMMUTABLE BORROWS
// ============================================================

mut v := Vec<int>()

a := v
b := v

print(a.len())
print(b.len())

v.push(1) // ERROR: mutable access while shared borrows exist


// ============================================================
// MUTABLE BORROWS
// ============================================================

mut v := Vec<int>()

f(mut v) // mutable borrow

v.push(1) // OK after call returns


fn f(mut x Vec<int>) {
    x.push(1)
}


// ============================================================
// IMMUTABLE BORROW PARAMS
// ============================================================

fn print_len(v Vec<int>) {
    print(v.len())
}

mut x := Vec<int>()

print_len(x) // shared immutable borrow

x.push(1) // OK after call returns


// ============================================================
// OWNERSHIP TRANSFER
// ============================================================

fn consume(own v Vec<int>) {
}

mut x := Vec<int>()

consume(own x)

x.push(1) // ERROR: moved value


// ============================================================
// MOVE INTO IMMUTABLE BINDING
// ============================================================

mut x := Vec<int>()

y := own x

x.push(1) // ERROR: moved
y.push(1) // ERROR: immutable binding


// ============================================================
// MOVE INTO MUTABLE BINDING
// ============================================================

x := Vec<int>()

mut y := own x

x.len() // ERROR: moved
y.push(1) // OK


// ============================================================
// SHARED VALUE CANNOT BECOME MUTABLE
// ============================================================

x := Vec<int>()

a := x
b := x

mut y := own x // ERROR: x is shared


// ============================================================
// UNIQUE IMMUTABLE CAN BECOME MUTABLE
// ============================================================

x := Vec<int>()

mut y := own x // OK

y.push(1)


// ============================================================
// REASSIGNMENT
// ============================================================

mut x := Vec<int>()
mut y := Vec<int>()

x = own y // OK


// ============================================================
// REASSIGNMENT WHILE BORROWED
// ============================================================

mut x := Vec<int>()

a := x

x = Vec<int>() // ERROR: reassignment while shared borrow active


// ============================================================
// MULTIPLE SHARED BORROWS
// ============================================================

mut x := Vec<int>()

a := x
b := x
c := x

print(a.len())
print(b.len())
print(c.len())


// ============================================================
// SHARED + MUTABLE BORROW CONFLICT
// ============================================================

mut x := Vec<int>()

a := x

f(mut x) // ERROR: mutable borrow while shared borrow active


// ============================================================
// MULTIPLE MUTABLE BORROWS
// ============================================================

mut x := Vec<int>()

f(mut x)
g(mut x) // ERROR if first mutable borrow still active


// ============================================================
// BORROW SCOPES
// ============================================================

mut x := Vec<int>()

{
    a := x
    print(a.len())
}

x.push(1) // OK


// ============================================================
// CONDITIONAL FLOW BORROWS
// ============================================================

mut x := Vec<int>()

if cond {
    a := x
}

x.push(1) // OK


// ============================================================
// CONDITIONAL CONFLICT
// ============================================================

mut x := Vec<int>()

a := x

if cond {
    x.push(1) // ERROR
}


// ============================================================
// LOOP BORROW
// ============================================================

mut x := Vec<int>()

for i := 0; i < 10; i = i + 1 {
    a := x
}

x.push(1) // OK


// ============================================================
// ITERATOR INVALIDATION
// ============================================================

mut v := Vec<int>()

it := v.iter()

v.push(1) // ERROR


// ============================================================
// NESTED MUTABLE BORROWS
// ============================================================

fn g(mut x Vec<int>) {
}

fn f(mut x Vec<int>) {
    g(mut x) // OK
}

mut v := Vec<int>()

f(mut v)


// ============================================================
// OVERLAPPING MUTABLE BORROWS
// ============================================================

fn g(mut x Vec<int>) {
}

mut v := Vec<int>()

g(mut v)
g(mut v) // OK sequentially


// ============================================================
// RETURNING SHARED REFERENCES
// ============================================================

fn f(v Vec<int>) => Vec<int> {
    return v
}

mut x := Vec<int>()

y := f(x)

x.push(1) // ERROR while y alive


// ============================================================
// COPY TYPES INSIDE REFERENCE TYPES
// ============================================================

Vec2 :: struct {
    x int
    y int
}

mut v := Vec<Vec2>()

p := v[0] // copies Vec2

p.x = 10 // OK on local copy


// ============================================================
// INDEXING REFERENCE TYPES
// ============================================================

mut v := Vec<Vec<int>>()

a := v[0]

v.push(Vec<int>()) // ERROR: shared borrow active


// ============================================================
// MUTABLE INDEX ACCESS
// ============================================================

mut v := Vec<Vec<int>>()

mut a := mut v[0]

a.push(1)


// ============================================================
// CLOSURE SHARED CAPTURE
// ============================================================

x := Vec<int>()

f := fn() {
    print(x.len())
}

print(x.len()) // OK


// ============================================================
// CLOSURE MOVE CAPTURE
// ============================================================

mut x := Vec<int>()

f := fn() {
    x.push(1)
}

x.push(1) // ERROR: moved into closure


// ============================================================
// ESCAPING CLOSURE
// ============================================================

fn make() => Fn {
    mut x := Vec<int>()

    return fn() {
        x.push(1)
    }
}


// ============================================================
// ASYNC BORROW VIOLATION
// ============================================================

async fn f(mut x Vec<int>) {
    await foo()

    x.push(1) // ERROR
}


// ============================================================
// ASYNC OWNERSHIP
// ============================================================

async fn f(own x Vec<int>) {
    await foo()

    x.push(1) // OK
}


// ============================================================
// CHANNEL OWNERSHIP TRANSFER
// ============================================================

mut x := Vec<int>()

channel.send(own x)

x.push(1) // ERROR


// ============================================================
// METHOD RECEIVER MUTABILITY
// ============================================================

x := Vec<int>()

x.push(1) // ERROR


mut x := Vec<int>()

x.push(1) // OK


// ============================================================
// METHOD SHARED BORROW
// ============================================================

mut x := Vec<int>()

a := x

x.len() // OK
x.push(1) // ERROR


// ============================================================
// STRUCT FIELD MUTABILITY
// ============================================================

User :: struct {
    name String
}

u := User{name:"a"}

u.name = "b" // ERROR


mut u := User{name:"a"}

u.name = "b" // OK


// ============================================================
// MUTABLE FIELD THROUGH SHARED ALIAS
// ============================================================

mut u := User{name:"a"}

a := u

u.name = "b" // ERROR


// ============================================================
// OWNERSHIP THROUGH STRUCTS
// ============================================================

Wrapper :: struct {
    data Vec<int>
}

mut w := Wrapper{
    data: Vec<int>()
}

x := own w

w.data.push(1) // ERROR


// ============================================================
// ARC SHARING
// ============================================================

x := Vec<int>()

a := x
b := x
c := x

// ARC count internally increased


// ============================================================
// ARC MOVE
// ============================================================

mut x := Vec<int>()

mut y := own x

// ARC count unchanged


// ============================================================
// COPY CONTAINING REFERENCES
// ============================================================

Handle :: struct {
    vec Vec<int>
}

h := Handle{
    vec: Vec<int>()
}

k := h // shallow copy + ARC retain

h.vec.len()
k.vec.len()


// ============================================================
// EQUALITY
// ============================================================

x := 10
y := 10

x == y // value equality


a := Vec<int>()
b := a

a == b // identity equality


// ============================================================
// TEMPORARY BORROWS
// ============================================================

mut x := Vec<int>()

print(x.len())

x.push(1) // OK


// ============================================================
// BORROW ENDS AFTER CALL
// ============================================================

fn inspect(v Vec<int>) {
}

mut x := Vec<int>()

inspect(x)

x.push(1) // OK


// ============================================================
// MOVE AFTER BORROW
// ============================================================

mut x := Vec<int>()

inspect(x)

y := own x // OK


// ============================================================
// BORROW AFTER MOVE
// ============================================================

mut x := Vec<int>()

y := own x

inspect(x) // ERROR


// ============================================================
// SELF ASSIGNMENT
// ============================================================

mut x := Vec<int>()

x = own x // ERROR


// ============================================================
// PARTIAL FIELD BORROW
// ============================================================

Pair :: struct {
    left Vec<int>
    right Vec<int>
}

mut p := Pair{
    left: Vec<int>(),
    right: Vec<int>()
}

a := p.left

p.right.push(1) // allowed only if borrow checker supports field splitting


// ============================================================
// WHOLE OBJECT CONFLICT
// ============================================================

mut p := Pair{
    left: Vec<int>(),
    right: Vec<int>()
}

a := p

p.right.push(1) // ERROR


// ============================================================
// RETURN MOVE
// ============================================================

fn make() => Vec<int> {
    mut x := Vec<int>()

    return own x
}


// ============================================================
// DOUBLE MOVE
// ============================================================

mut x := Vec<int>()

y := own x
z := own x // ERROR


// ============================================================
// MOVE THEN REINITIALIZE
// ============================================================

mut x := Vec<int>()

y := own x

x = Vec<int>() // OK

x.push(1)


// ============================================================
// MOVE INTO ASYNC TASK
// ============================================================

mut x := Vec<int>()

spawn worker(own x)

x.push(1) // ERROR
```
